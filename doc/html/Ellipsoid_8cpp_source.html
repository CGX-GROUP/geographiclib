00068   Math::real Ellipsoid::InverseRectifyingLatitude(real mu) const throw() {
00069     if (abs(mu) == 90)
00070       return mu;
00071     // mu depends on beta nearly linearly (much more so that it depends on
00072     // phi).  So find beta first by solving E(beta, -e12) = e0 with initial
00073     // guess beta = mu
00074     real
00075       e0 = mu * QuarterMeridian() / (90 * _b),
00076       beta = mu;
00077     for (int i = 0; i < numit_; ++i) {
00078       real err = _ell.Ed(beta) - e0;
00079       beta = beta -
00080         err / ( sqrt(1 + _e12 * Math::sq(sin(beta *  Math::degree<real>())))
00081                 * Math::degree<real>());
00082       if (abs(err) < _stol)
00083         break;
00084     }
00085     return InverseParametricLatitude(beta);
00086   }
