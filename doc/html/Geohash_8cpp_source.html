00032   void Geohash::Forward(real lat, real lon, int len, std::string& geohash) {
00033     if (lat < -90 || lat > 90)
00034       throw GeographicErr("Latitude " + Utility::str(lat)
00035                           + "d not in [-90d, 90d]");
00036     if (lon < -180 || lon > 360)
00037       throw GeographicErr("Longitude " + Utility::str(lon)
00038                           + "d not in [-180d, 360d]");
00039     if (lat == 90) lat -= lateps_ / 2;
00040     if (lon >= 180) lon -= 360; // lon in [-180,180)
00041     // lon/loneps_ in [-2^45,2^45); lon/eps + shift_ in [0,2^46)
00042     // similarly for lat
00043     len = max(0, min(int(maxlen_), len));
00044     unsigned long long
00045       ulon = (unsigned long long)(floor(lon/loneps_) + shift_),
00046       ulat = (unsigned long long)(floor(lat/lateps_) + shift_);
00047     char geohash1[maxlen_];
00048     unsigned byte = 0;
00049     for (unsigned i = 0; i < 5 * unsigned(len);) {
00050       if ((i & 1) == 0) {
00051         byte = (byte << 1) + unsigned((ulon & mask_) != 0);
00052         ulon <<= 1;
00053       } else {
00054         byte = (byte << 1) + unsigned((ulat & mask_) != 0);
00055         ulat <<= 1;
00056       }
00057       ++i;
00058       if (i % 5 == 0) {
00059         geohash1[(i/5)-1] = lcdigits_[byte];
00060         byte = 0;
00061       }
00062     }
00063     geohash.resize(len);
00064     copy(geohash1, geohash1 + len, geohash.begin());
